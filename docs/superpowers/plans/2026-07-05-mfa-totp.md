# TOTP MFA (step-up) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add TOTP two-factor auth for local password accounts — self-service enrolment (QR + 8 backup codes), a two-step stateless-challenge login, and an admin "require MFA" policy toggle.

**Architecture:** TOTP via `pquerna/otp` (a thin `services/auth/internal/mfa` wrapper). The secret is AES-256-GCM encrypted at rest under a dedicated `MFA_SECRET_KEY_HEX` KEK. Login becomes two-step via short-lived, `typ`+audience-scoped RS256 tokens (no new server store). Enrolment routes live on the auth HTTP handler (like `/users/me/*`); the admin toggle rides `token_policies`. MFA state is read via dedicated repository methods so the shared `User`/`scanOne` row is untouched. SSO users are exempt.

**Tech Stack:** Go (`services/auth`), `github.com/pquerna/otp`, `libs/crypto/aes` + `libs/crypto/argon2`, PostgreSQL (goose), Redis (existing lockout), React + TanStack Query + `qrcode.react` (`frontend`).

**Design doc:** `docs/superpowers/specs/2026-07-05-mfa-totp-design.md`. Read it before starting.

**Conventions:**
- Branch `feat/mfa-totp` is checked out. Commit after every green task (per §15.2 run `cd services/auth && go build ./... && go vet ./... && go test ./...` before pushing; frontend tasks run the 4 gates in §15.1).
- Integration tests are `//go:build integration` + testcontainers; the standard `go test ./...` skips them. Run them with `go test -tags integration ./...` when Docker is up.
- All new code gets comments (project rule). Never log secrets, `otpauth_uri`, backup codes, or OTP inputs.
- Migration timestamps: use a real `YYYYMMDDHHMMSS` after the latest existing (`20260703120300`), e.g. `20260705120000`. Pick the actual current time at implementation; keep it unique and later than every existing auth migration.

```go
// Package mfa wraps pquerna/otp to provide the TOTP primitives the auth
// service needs for two-factor auth: minting an enrolment secret + otpauth URI,
// validating a code against a ±1 time-step window while returning the matched
// counter (for replay prevention), and generating single-use backup codes.
// This package holds NO database or HTTP logic.
package mfa

import (
	"crypto/rand"
	"crypto/subtle"
	"encoding/base32"
	"fmt"
	"time"

	"github.com/pquerna/otp"
	"github.com/pquerna/otp/hotp"
	"github.com/pquerna/otp/totp"
)

const (
	// periodSeconds is the TOTP time step (RFC 6238 default).
	periodSeconds = 30
	// secretBytes is the raw shared-secret length (160-bit, RFC 6238 minimum).
	secretBytes = 20
	// digits is the OTP length.
	digits = otp.DigitsSix
	// skewSteps is how many periods on each side of "now" are accepted, to
	// tolerate client/server clock drift.
	skewSteps = 1
	// backupCodeCount is how many single-use recovery codes an enrolment mints.
	backupCodeCount = 8
	// backupCodeBytes is the entropy per backup code before base32 encoding.
	backupCodeBytes = 10
)

// GenerateSecret mints a fresh 20-byte base32 TOTP secret plus its otpauth://
// provisioning URI (issuer + account label). The FE renders a QR from the URI.
// The returned secret is the value stored (encrypted) at rest.
func GenerateSecret(issuer, account string) (secretBase32, otpauthURI string, err error) {
	key, err := totp.Generate(totp.GenerateOpts{
		Issuer:      issuer,
		AccountName: account,
		SecretSize:  secretBytes,
		Digits:      digits,
		Period:      periodSeconds,
	})
	if err != nil {
		return "", "", fmt.Errorf("generate totp secret: %w", err)
	}
	return key.Secret(), key.URL(), nil
}

// ValidateCode reports whether code is a valid TOTP for secretBase32 at time t,
// within ±skewSteps. On success it returns the matched time-step counter so the
// caller can enforce replay prevention (reject counters ≤ the last accepted).
// The comparison is constant-time.
func ValidateCode(secretBase32, code string, t time.Time) (ok bool, counter int64) {
	base := uint64(t.Unix()) / periodSeconds
	for delta := -skewSteps; delta <= skewSteps; delta++ {
		c := base + uint64(delta) //nolint:gosec // bounded by skewSteps
		want, err := hotp.GenerateCodeCustom(secretBase32, c, hotp.ValidateOpts{
			Digits:    digits,
			Algorithm: otp.AlgorithmSHA1,
		})
		if err != nil {
			continue
		}
		if subtle.ConstantTimeCompare([]byte(want), []byte(code)) == 1 {
			return true, int64(c)
		}
	}
	return false, 0
}

// GenerateBackupCodes returns backupCodeCount distinct base32 recovery codes.
// Callers argon2-hash these before storage and surface the plaintext to the
// user exactly once.
func GenerateBackupCodes() ([]string, error) {
	enc := base32.StdEncoding.WithPadding(base32.NoPadding)
	out := make([]string, 0, backupCodeCount)
	seen := make(map[string]struct{}, backupCodeCount)
	for len(out) < backupCodeCount {
		buf := make([]byte, backupCodeBytes)
		if _, err := rand.Read(buf); err != nil {
			return nil, fmt.Errorf("read random for backup code: %w", err)
		}
		code := enc.EncodeToString(buf)
		if _, dup := seen[code]; dup {
			continue
		}
		seen[code] = struct{}{}
		out = append(out, code)
	}
	return out, nil
}
```

```go
package repository

import (
	"context"
	"errors"
	"fmt"
	"time"

	"github.com/google/uuid"
	"github.com/jackc/pgx/v5"
)

// MFAState is the per-user TOTP state, read on the login hot path and during
// enrolment. Kept separate from User so the shared scanOne row is untouched.
type MFAState struct {
	Enabled          bool
	SecretEnc        []byte     // AES-256-GCM ciphertext; nil when not enrolled
	SecretKEKVersion *int16     // rekey generation stamped on SecretEnc
	EnrolledAt       *time.Time
	LastUsedCounter  *int64     // TOTP counter of the last accepted code
}

// GetMFAState loads the user's MFA columns. Returns ErrNotFound if the user id
// does not exist.
func (r *UserRepository) GetMFAState(ctx context.Context, userID uuid.UUID) (*MFAState, error) {
	const q = `
		SELECT mfa_enabled, mfa_secret_enc, mfa_secret_kek_version,
		       mfa_enrolled_at, mfa_last_used_counter
		FROM   users
		WHERE  id = $1`
	var m MFAState
	err := r.pool.QueryRow(ctx, q, userID).Scan(
		&m.Enabled, &m.SecretEnc, &m.SecretKEKVersion, &m.EnrolledAt, &m.LastUsedCounter,
	)
	if err != nil {
		if errors.Is(err, pgx.ErrNoRows) {
			return nil, ErrNotFound
		}
		return nil, fmt.Errorf("get mfa state: %w", err)
	}
	return &m, nil
}

// SetPendingMFASecret stores an encrypted secret with mfa_enabled left false
// (enrolment step 1). Overwrites any prior pending secret.
func (r *UserRepository) SetPendingMFASecret(ctx context.Context, userID uuid.UUID, secretEnc []byte, kekVersion int16) error {
	const q = `
		UPDATE users
		SET mfa_secret_enc = $1, mfa_secret_kek_version = $2,
		    mfa_enabled = false, mfa_last_used_counter = NULL, updated_at = now()
		WHERE id = $3`
	return r.execAffectingOne(ctx, "set pending mfa secret", q, secretEnc, kekVersion, userID)
}

// EnableMFA flips mfa_enabled true and stamps mfa_enrolled_at (enrolment step 2).
func (r *UserRepository) EnableMFA(ctx context.Context, userID uuid.UUID) error {
	const q = `UPDATE users SET mfa_enabled = true, mfa_enrolled_at = now(), updated_at = now() WHERE id = $1`
	return r.execAffectingOne(ctx, "enable mfa", q, userID)
}

// DisableMFA clears all MFA state on the user row (backup codes are deleted
// separately by the caller / ON DELETE CASCADE is not triggered here).
func (r *UserRepository) DisableMFA(ctx context.Context, userID uuid.UUID) error {
	const q = `
		UPDATE users
		SET mfa_enabled = false, mfa_secret_enc = NULL, mfa_secret_kek_version = NULL,
		    mfa_enrolled_at = NULL, mfa_last_used_counter = NULL, updated_at = now()
		WHERE id = $1`
	return r.execAffectingOne(ctx, "disable mfa", q, userID)
}

// AdvanceMFACounter records the counter of the last accepted TOTP code so the
// same code cannot be replayed.
func (r *UserRepository) AdvanceMFACounter(ctx context.Context, userID uuid.UUID, counter int64) error {
	const q = `UPDATE users SET mfa_last_used_counter = $1, updated_at = now() WHERE id = $2`
	return r.execAffectingOne(ctx, "advance mfa counter", q, counter, userID)
}

// InsertBackupCodes replaces the user's backup codes with the given argon2
// hashes in one transaction (delete-then-insert). Used on enrol + regenerate.
func (r *UserRepository) InsertBackupCodes(ctx context.Context, userID uuid.UUID, hashes []string) error {
	tx, err := r.pool.Begin(ctx)
	if err != nil {
		return fmt.Errorf("begin backup codes tx: %w", err)
	}
	defer func() { _ = tx.Rollback(ctx) }()
	if _, err := tx.Exec(ctx, `DELETE FROM user_mfa_backup_codes WHERE user_id = $1`, userID); err != nil {
		return fmt.Errorf("clear backup codes: %w", err)
	}
	for _, h := range hashes {
		if _, err := tx.Exec(ctx,
			`INSERT INTO user_mfa_backup_codes (user_id, code_hash) VALUES ($1, $2)`, userID, h); err != nil {
			return fmt.Errorf("insert backup code: %w", err)
		}
	}
	return tx.Commit(ctx)
}

// DeleteBackupCodes removes all of the user's backup codes (on disable).
func (r *UserRepository) DeleteBackupCodes(ctx context.Context, userID uuid.UUID) error {
	_, err := r.pool.Exec(ctx, `DELETE FROM user_mfa_backup_codes WHERE user_id = $1`, userID)
	if err != nil {
		return fmt.Errorf("delete backup codes: %w", err)
	}
	return nil
}

// BackupCode is one stored recovery code (hash + used state).
type BackupCode struct {
	ID       uuid.UUID
	CodeHash string
	Used     bool
}

// ListUnusedBackupCodes returns the user's not-yet-consumed codes for the
// service to argon2-compare against a submitted code.
func (r *UserRepository) ListUnusedBackupCodes(ctx context.Context, userID uuid.UUID) ([]BackupCode, error) {
	const q = `SELECT id, code_hash FROM user_mfa_backup_codes WHERE user_id = $1 AND used_at IS NULL`
	rows, err := r.pool.Query(ctx, q, userID)
	if err != nil {
		return nil, fmt.Errorf("list backup codes: %w", err)
	}
	defer rows.Close()
	var out []BackupCode
	for rows.Next() {
		var c BackupCode
		if err := rows.Scan(&c.ID, &c.CodeHash); err != nil {
			return nil, fmt.Errorf("scan backup code: %w", err)
		}
		out = append(out, c)
	}
	return out, rows.Err()
}

// MarkBackupCodeUsed stamps used_at on a single code, but only if still unused
// (guards against a concurrent double-spend). Returns ErrNotFound if already used.
func (r *UserRepository) MarkBackupCodeUsed(ctx context.Context, id uuid.UUID) error {
	const q = `UPDATE user_mfa_backup_codes SET used_at = now() WHERE id = $1 AND used_at IS NULL`
	return r.execAffectingOne(ctx, "mark backup code used", q, id)
}

// execAffectingOne runs an UPDATE/DELETE expected to touch exactly one row,
// mapping zero rows to ErrNotFound.
func (r *UserRepository) execAffectingOne(ctx context.Context, label, q string, args ...any) error {
	tag, err := r.pool.Exec(ctx, q, args...)
	if err != nil {
		return fmt.Errorf("%s: %w", label, err)
	}
	if tag.RowsAffected() == 0 {
		return ErrNotFound
	}
	return nil
}
```

```go
// mfaKEKVersion is the KEK generation stamped on freshly-encrypted MFA secrets.
// Rotation to a new version is handled out-of-band by the rotate-kek tool.
const mfaKEKVersion int16 = 1

// ErrMFAAlreadyEnabled is returned when enrolling a user who already has MFA on.
var ErrMFAAlreadyEnabled = errors.New("mfa already enabled")

// ErrMFANotEnrolled is returned when verifying/using MFA for a user with none.
var ErrMFANotEnrolled = errors.New("mfa not enrolled")

// MFAStatus is the self-service status payload.
type MFAStatus struct {
	Enabled    bool
	EnrolledAt *string // RFC3339, nil when not enrolled
}

// GetMFAStatus reports whether the user has MFA enabled.
func (s *Service) GetMFAStatus(ctx context.Context, userID uuid.UUID) (MFAStatus, error) {
	st, err := s.users.GetMFAState(ctx, userID)
	if err != nil {
		return MFAStatus{}, err
	}
	out := MFAStatus{Enabled: st.Enabled}
	if st.EnrolledAt != nil {
		v := st.EnrolledAt.UTC().Format("2006-01-02T15:04:05Z07:00")
		out.EnrolledAt = &v
	}
	return out, nil
}

// BeginMFAEnrollment mints a fresh secret, stores it encrypted (pending), and
// returns the base32 secret + otpauth URI for the QR. Rejects if already on.
func (s *Service) BeginMFAEnrollment(ctx context.Context, userID uuid.UUID, account string) (secretBase32, otpauthURI string, err error) {
	st, err := s.users.GetMFAState(ctx, userID)
	if err != nil {
		return "", "", err
	}
	if st.Enabled {
		return "", "", ErrMFAAlreadyEnabled
	}
	secret, uri, err := mfa.GenerateSecret(s.mfaIssuer, account)
	if err != nil {
		return "", "", err
	}
	enc, err := aespkg.Encrypt([]byte(secret), s.mfaKEK)
	if err != nil {
		return "", "", fmt.Errorf("encrypt mfa secret: %w", err)
	}
	if err := s.users.SetPendingMFASecret(ctx, userID, enc, mfaKEKVersion); err != nil {
		return "", "", err
	}
	return secret, uri, nil
}

// CompleteMFAEnrollment verifies the first code against the pending secret; on
// success it enables MFA, generates + stores 8 argon2-hashed backup codes, and
// returns the plaintext codes once.
func (s *Service) CompleteMFAEnrollment(ctx context.Context, userID uuid.UUID, code string) ([]string, error) {
	st, err := s.users.GetMFAState(ctx, userID)
	if err != nil {
		return nil, err
	}
	if len(st.SecretEnc) == 0 {
		return nil, ErrMFANotEnrolled
	}
	ok, err := s.verifyTOTP(ctx, userID, st, code)
	if err != nil {
		return nil, err
	}
	if !ok {
		return nil, ErrInvalidCredentials
	}
	if err := s.users.EnableMFA(ctx, userID); err != nil {
		return nil, err
	}
	return s.regenerateBackupCodes(ctx, userID)
}

// verifyTOTP decrypts the secret, validates the code within the skew window,
// enforces replay prevention (counter must advance), and persists the counter.
func (s *Service) verifyTOTP(ctx context.Context, userID uuid.UUID, st *repository.MFAState, code string) (bool, error) {
	secret, err := aespkg.Decrypt(st.SecretEnc, s.mfaKEK)
	if err != nil {
		return false, fmt.Errorf("decrypt mfa secret: %w", err)
	}
	ok, counter := mfa.ValidateCode(string(secret), code, s.now())
	if !ok {
		return false, nil
	}
	if st.LastUsedCounter != nil && counter <= *st.LastUsedCounter {
		return false, nil // replay of an already-used code
	}
	if err := s.users.AdvanceMFACounter(ctx, userID, counter); err != nil {
		return false, err
	}
	return true, nil
}

// regenerateBackupCodes mints, hashes, and stores 8 fresh codes, returning the
// plaintext once.
func (s *Service) regenerateBackupCodes(ctx context.Context, userID uuid.UUID) ([]string, error) {
	codes, err := mfa.GenerateBackupCodes()
	if err != nil {
		return nil, err
	}
	hashes := make([]string, len(codes))
	for i, c := range codes {
		h, herr := argon2pkg.Hash(c)
		if herr != nil {
			return nil, fmt.Errorf("hash backup code: %w", herr)
		}
		hashes[i] = h
	}
	if err := s.users.InsertBackupCodes(ctx, userID, hashes); err != nil {
		return nil, err
	}
	return codes, nil
}
```

```go
// ConsumeMFACode validates a submitted code as EITHER a TOTP (with replay
// prevention) OR an unused backup code (single-use). Returns true on success.
func (s *Service) ConsumeMFACode(ctx context.Context, userID uuid.UUID, code string) (bool, error) {
	st, err := s.users.GetMFAState(ctx, userID)
	if err != nil {
		return false, err
	}
	if !st.Enabled || len(st.SecretEnc) == 0 {
		return false, ErrMFANotEnrolled
	}
	// TOTP first.
	ok, err := s.verifyTOTP(ctx, userID, st, code)
	if err != nil {
		return false, err
	}
	if ok {
		return true, nil
	}
	// Then backup codes.
	codes, err := s.users.ListUnusedBackupCodes(ctx, userID)
	if err != nil {
		return false, err
	}
	for _, bc := range codes {
		match, verr := argon2pkg.Verify(code, bc.CodeHash)
		if verr == nil && match {
			// Single-use: mark consumed; a concurrent double-spend loses the race.
			if merr := s.users.MarkBackupCodeUsed(ctx, bc.ID); merr != nil {
				return false, nil // already used by a racing request
			}
			return true, nil
		}
	}
	return false, nil
}
```

```go
		return

	var req struct {

			writeError(w, http.StatusBadRequest, "MFACODEINVALID", "invalid code")
			return
		}
		slog.ErrorContext(r.Context(), "mfa verify failed", "error", err)
		writeError(w, http.StatusInternalServerError, "INTERNAL", "internal error")
		return
	}
	resp := map[string]any{"backup_codes": codes}
	// Forced-enrolment: a setup-token-driven verify completes login by also
	// returning a full access token (amr=["pwd","otp"]).

}
```

```go
// LoginResult is the outcome of a password login. Exactly one of the three
// states is set: an access Token, an MFA challenge (MFARequired), or a
// forced-enrolment setup (MFASetupRequired).
type LoginResult struct {
	Token            string
	MFARequired      bool
	ChallengeToken   string
	MFASetupRequired bool
	SetupToken       string
}

func (s *Service) Login(ctx context.Context, tenantID uuid.UUID, username, password string) (LoginResult, error) {
	user, err := s.AuthenticateUser(ctx, tenantID, username, password)
	if err != nil {
		return LoginResult{}, err
	}
	mfaState, err := s.users.GetMFAState(ctx, user.ID)
	if err != nil {
		return LoginResult{}, err
	}
	if mfaState.Enabled {
		ct, terr := s.IssueMFAChallengeToken(ctx, user.ID.String(), user.TenantID.String())
		if terr != nil {
			return LoginResult{}, terr
		}
		return LoginResult{MFARequired: true, ChallengeToken: ct}, nil
	}
	// Forced enrolment: policy requires MFA and this password user has none.
	policy, perr := s.tokenPolicies.GetOrDefault(ctx, user.TenantID)
	if perr == nil && policy.RequireMFA && user.Kind == "human" {
		stk, terr := s.IssueMFASetupToken(ctx, user.ID.String(), user.TenantID.String())
		if terr != nil {
			return LoginResult{}, terr
		}
		return LoginResult{MFASetupRequired: true, SetupToken: stk}, nil
	}
	roles := s.loadRoleNames(ctx, user.ID, user.TenantID)
	tok, terr := s.IssueToken(ctx, user.ID.String(), user.TenantID.String(), nil, roles, user.IsGlobalAdmin, user.Kind, []string{"pwd"})
	if terr != nil {
		return LoginResult{}, terr
	}
	return LoginResult{Token: tok}, nil
}
```

```ts
export type LoginResult =
  | { kind: "token" }
  | { kind: "mfa"; challengeToken: string }
  | { kind: "mfa_setup"; setupToken: string };

export async function login(username: string, password: string, tenantId: string): Promise<LoginResult> {
  const { data } = await apiClientRaw.post<{
    token?: string; session_token?: string;
    mfa_required?: boolean; challenge_token?: string;
    mfa_setup_required?: boolean; setup_token?: string;
  }>("/login", { tenant_id: tenantId, username, password });

  if (data.mfa_required && data.challenge_token) return { kind: "mfa", challengeToken: data.challenge_token };
  if (data.mfa_setup_required && data.setup_token) return { kind: "mfa_setup", setupToken: data.setup_token };

  const token = data.token ?? data.session_token;
  if (!token) throw new Error("Login succeeded but no token was returned.");
  authStore.setToken(token);
  return { kind: "token" };
}

// Complete the OTP step; stores the returned access token.
export async function loginMfa(challengeToken: string, code: string): Promise<void> {
  const { data } = await apiClientRaw.post<{ token: string }>("/login/mfa", {
    challenge_token: challengeToken, code,
  });
  authStore.setToken(data.token);
}
```

## Notes for the implementer

- **Do not** add MFA fields to the shared `User` struct / `scanOne` — use the dedicated `MFAState` reads (Task 3). Touching `scanOne` means updating ~7 SELECT lists.
- **`IssueToken` signature changed** (added `amr`). Grep every caller and pass the right `amr` (`["pwd"]` password, `["pwd","otp"]` MFA, `["sso"]` SSO, `nil` service/other).
- **Never** log the secret, `otpauth_uri`, backup codes, or OTP inputs (§10). The enrol/verify responses carry secrets — they're returned to the authenticated owner over HTTPS only, never logged.
- **SSO users are exempt** — the MFA branch in `Login` only runs on the password path; the SSO callback path is untouched.
- **`RefreshToken`** must pass `claims.Amr` to `IssueToken` and must never be reachable with a challenge/setup token (guaranteed because it calls `ValidateToken`, which now rejects typed tokens).
